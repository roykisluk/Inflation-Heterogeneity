Demographic groups¶

In [3]:
# Count rows for each group
group_counts = {group: sum(len(data[year]) for year in data) for group, data in demo.items()}

# Plot
plt.figure(figsize=(10, 6))
plt.bar(group_counts.keys(), group_counts.values(), color='skyblue')
plt.xlabel('Groups')
plt.ylabel('Number of Observations')
plt.title('Number of Observations for Each Group')
plt.xticks(rotation=45, ha='right')
plt.tight_layout()
plt.show()
